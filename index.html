🗄️ Migración Supabase — Lote / Referencia / Caducidad
Para que estos campos se guarden en la nube y sincronicen entre dispositivos, ejecuta este SQL en Supabase → SQL Editor:
-- Columnas extra stock ALTER TABLE stock_shirake ADD COLUMN IF NOT EXISTS lote text DEFAULT ''; ALTER TABLE stock_shirake ADD COLUMN IF NOT EXISTS referencia text DEFAULT ''; ALTER TABLE stock_shirake ADD COLUMN IF NOT EXISTS caducidad text DEFAULT ''; ALTER TABLE stock_shirake ADD COLUMN IF NOT EXISTS codigo text DEFAULT ''; -- Columnas contactos ALTER TABLE contactos_shirake ADD COLUMN IF NOT EXISTS nif text DEFAULT ''; ALTER TABLE contactos_shirake ADD COLUMN IF NOT EXISTS persona text DEFAULT ''; ALTER TABLE contactos_shirake ADD COLUMN IF NOT EXISTS direccion text DEFAULT ''; -- Tiempo real (ejecutar una sola vez) ALTER PUBLICATION supabase_realtime ADD TABLE pedidos_shirake; ALTER PUBLICATION supabase_realtime ADD TABLE stock_shirake; ALTER PUBLICATION supabase_realtime ADD TABLE contactos_shirake;
